components/Form/PasswordField.js

// @ts-check

import React, { useState } from 'react';
import { View, TextInput, TouchableOpacity, Text } from 'react-native';
import '../../typedefs';
import styles from './styles';
import { CloseIcon } from '../SvgIcons';
import { Keyboard } from 'react-native';

/**
 * Renders Password input
 * @param {Object} props - Props pass to PasswordField
 * @param {string} props.value - Value of the PasswordField
 * @param {function} props.onChangeText - On text change handler
 * @param {Object=} props.ref - Props pass to PasswordField
 * @param {Object=} props.style - Props pass to PasswordField
 * @param {Object=} props.placeholder - Placeholder of the PasswordField
 * @param {callbackFunction=} props.onSubmitEditing - Callback that is called when the text input's submit button is pressed with the argument
 * @param {callbackFunction=} props.onBlur - On blur handler for PasswordField
 * @param {boolean=} props.clearable - With clear button
 * @param {boolean=} props.showPassword - With show password button
 */
const PasswordField = props => {
  const {
    ref,
    style,
    value,
    placeholder,
    onChangeText,
    onSubmitEditing,
    onBlur,
    clearable,
    showPassword,
  } = props;

  const [passwordVisibility, togglePasswordVisibility] = useState(true);
  const clearPassword = () => {
    onChangeText('');
  };

  return (
    <View style={style}>
      <TextInput
        style={styles.textField}
        placeholder={placeholder}
        value={value}
        placeholderTextColor="#BFBFBF"
        onChangeText={text => onChangeText(text)}
        secureTextEntry={passwordVisibility}
        returnKeyType="done"
        ref={ref}
        onBlur={onBlur}
        autoCompleteType="password"
        blurOnSubmit={false}
        onSubmitEditing={() => {
          onSubmitEditing();
          Keyboard.dismiss();
        }}
      />
      {showPassword && (
        <TouchableOpacity
          style={styles.showPassword}
          onPress={() => togglePasswordVisibility(!passwordVisibility)}>
          <Text>{passwordVisibility ? 'Show' : 'Hide'}</Text>
        </TouchableOpacity>
      )}
      {clearable && (
        <TouchableOpacity style={styles.clearPassword} onPress={clearPassword}>
          <CloseIcon width={6} height={6} color="#fff" />
        </TouchableOpacity>
      )}
    </View>
  );
};

export default PasswordField;
