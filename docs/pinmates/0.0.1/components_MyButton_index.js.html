// @ts-check
import React from 'react';

import { TouchableOpacity, View } from 'react-native';
import LinearGradient from 'react-native-linear-gradient';
// @ts-ignore
import colors from '../../constants/colors';

import MyText from '../MyText';
import '../../typedefs';
import styles from './styles';

/**
 * Renders Button component
 * @param {Object} props - Props passed to Button
 * @param {string=} props.text - Text of Button
 * @param {string} props.color - Set color of background
 * @param {callbackFunction} props.onPress - Handles onPress on Button
 * @param {React.ReactNode=} props.customText - Pass custom text component
 * @param {boolean=} props.disabled - For checking if disabled
 * @param {boolean=} props.fullWidth - If width is 100%
 * @param {Object=} props.style - Additional styles applied on wrapper Container
 * @param {"lg"=} props.size - Size of Button
 * @param {function=} props.iconRight - Icon to be rendered on right
 * @param {function=} props.iconLeft - Icon to be rendered on left
 * @param {boolean=} props.rounded - If border is rounded
 * @param {boolean=} props.outline - Without background, only border
 * @param {boolean=} props.link - Without background and border
 * @param {boolean=} props.nonGradient - Set color of background without gradient
 */
const MyButton = ({
  text,
  color,
  onPress,
  customText,
  disabled,
  fullWidth,
  style,
  size,
  iconRight,
  iconLeft,
  rounded,
  outline,
  link,
  nonGradient,
}) => {
  const LeftIcon = () => {
    return <View style={styles.leftIcon}>{iconLeft()}</View>;
  };

  const RightIcon = () => {
    return <View style={styles.rightIcon}>{iconRight()}</View>;
  };

  const RenderContent = () => {
    return (
      <>
        {iconLeft && <LeftIcon />}
        {customText ? (
          customText
        ) : (
          <MyText
            type={size && size === 'lg' && 's20h24w500'}
            color={outline ? color : 'white'}>
            {text}
          </MyText>
        )}

        {iconRight && <RightIcon />}
      </>
    );
  };

  return (
    <TouchableOpacity disabled={disabled} onPress={onPress}>
      <LinearGradient
        colors={
          outline || link
            ? [colors.transparent]
            : nonGradient
            ? [colors[color], colors[color]]
            : colors[color]
        }
        style={[
          styles.default,
          style && style,
          size && size === 'lg' && styles.sizeLarge,
          rounded && styles.rounded,
          outline && [
            styles.outline,
            {
              borderColor: colors[color],
            },
          ],
          disabled && styles.disabled,
          fullWidth && styles.fullWidth,
        ]}>
        {RenderContent()}
      </LinearGradient>
    </TouchableOpacity>
  );
};

export default MyButton;
